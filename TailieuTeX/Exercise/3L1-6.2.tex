\begin{ex} \nguon{2}
	Người thợ rèn  nung chảy một miếng nhôm ban đầu ở nhiệt độ $60,3^\circ C$. Sau $18 \ s$, nhôm đạt đến nhiệt độ nóng chảy $660,3^\circ C$. Biết công suất của lò nung không đổi. Nhiệt nóng chảy riêng của nhôm là $390 \ kJ /kg$; nhiệt dung riêng của nhôm là $880 \ J / kg. K$. Thời gian để nung chảy hoàn toàn khối nhôm này khi nó đang ở nhiệt độ nóng chảy xấp xỉ là
	\choice
	{\True $13,3\ s$}
	{$17,5\ s$}
	{$20,0\ s$}
	{$10,7\ s$}
	\loigiai{
		\begin{itemize}
			\item Công suất của lò nung:
			\begin{align*}
				\mathcal{P}=\dfrac{A}{t}
			\end{align*}
			\item Công suất của lò nung không thay đổi, tức là chỉ có công $A$ và thời gian $t$ thay đổi. Do đó:
			\begin{align*}
				\mathcal{P}=\dfrac{Q_1}{t_1}=\dfrac{Q_2}{t_2}
			\end{align*}
			Trong đó 
			\begin{itemize}
				\item $Q_1$ là nhiệt lượng cung cấp cho miếng nhôm từ nhiệt độ $t_1=60,3^\circ C$ đến nhiệt độ $t_2=660,3^\circ$ C
				\item $Q_2$ là nhiệt lượng cung cấp cho miếng nhôm nóng chảy hoàn toàn ở nhiệt độ $t_2=660,3^\circ C$
			\end{itemize}
			\item Nhiệt lượng cung cấp cho miếng nhôm từ nhiệt độ $t_1=60,3^\circ C$ đến nhiệt độ $t_2=660,3^\circ C$ là:
			\begin{align*}
				Q_1=c m \Delta t=c m\left(t_2-t_1\right)
			\end{align*}
			\item Nhiệt lượng cung cấp cho miếng nhôm nóng chảy hoàn toàn ở nhiệt độ $t_2=660,3^\circ C$
			\begin{align*}
				Q_2=\lambda m
			\end{align*}
			\item Từ các dữ liệu đã có, tìm thời gian nóng chảy $t_2$:
			\begin{align*}
				t_2=\dfrac{Q_2 t_1}{Q_1}=\dfrac{\lambda m t_1}{c m\left(t_2-t_1\right)}=\dfrac{\lambda t_1}{c\left(t_2-t_1\right)}=\dfrac{390 . 10^3 . 18}{880 .\left(660,3-60,3\right)}\approx 13,3 \ s
			\end{align*}
		\end{itemize}
	}
\end{ex}
