\loai{Loại 1: Chuyển hóa công thành nội năng}

\begin{ex}
	Người ta cọ xát hai vật với nhau, nhiệt dung của hai vật bằng nhau và bằng $800\ J / K$. Sau 1 phút người ta thấy nhiệt độ của mỗi vật tăng thêm $30\ K$. Công suất trung bình của việc cọ xát bằng
	\choice
	{$1080\ W$}
	{$980\ W$}
	{$480\ W$}
	{\True $800\ W$}
	\loigiai{
		Toàn bộ công cọ xát chuyển hết thành nhiệt của 2 vật
		$$
		A=Q \Rightarrow P t=2 m c \Delta t \Rightarrow P . 60=2.800.30 \Rightarrow P=800\ W 
		$$ 
	}
	\haidong{6}\end{ex}

\loai{Loại 2: Chuyển hóa cơ năng thành nội năng}

\begin{ex}
	Trong một thí nghiệm, người ta thả rơi tự do một mảnh thép từ độ cao $500\ m$, khi tới sát mặt đất nó có tốc độ $50\ m/s$. Cho biết nhiệt dung riêng của thép $c=460\ J/kg.K$ và lấy $g=10\ m/s^2$. Mảnh thép đã nóng thêm bao nhiêu độ ngay trước khi chạm đất, nếu cho rằng toàn bộ công cản của không khí chỉ dùng để làm nóng mảnh thép?
	\choice
	{$9,25\ K$}
	{\True $8,15\ K$}
	{$7,15\ K$}
	{$9,18\ K$}
	\loigiai{
		$Q=W_1-W_2 \Rightarrow m c \Delta t=m g h-\dfrac{1}{2} m v^2 \Rightarrow \Delta t=\dfrac{g h-\dfrac{1}{2} v^2}{c}=\dfrac{10.500-\dfrac{1}{2}.50^2}{460} \approx 8,15\ K$.	
	}
	\haidong{6}\end{ex}

\begin{ex}
	\immini[thm]{Xác định độ biến thiên nhiệt độ của nước rơi từ độ cao $96\ m$ xuống và đập vào cánh tuabin làm quay máy phát điện, biết rằng $50 \%$ thế năng ban đầu của nước biến thành nội năng của nước. Cho biết nhiệt dung riêng của nước là $4190\ J/kg.K$. Chọn mốc thế năng tại mặt đất. Lấy $g=9,8\ m/s^2$.
	}{\vspace*{1cm}\includegraphics[width=8cm]{images/5118}} \choice
	{$1,25\ K$}
	{$1,42\ K$}
	{\True $0,11\ K$}
	{$0,18\ K$}
	\loigiai{
		$Q=H W \Rightarrow m c \Delta t=H. m g h \Rightarrow \Delta t=\dfrac{H g h}{c}=\dfrac{0,5.9,8.96}{4190} \approx 0,11\ K$
	}
	\haidong{5}\end{ex}

\begin{ex} \nguon{2}
	\immini[thm]{Thác nước thiên thần ở Venezuela có chiều cao nước đổ là $807 \ m$, là thác cao nhất thế giới. Nếu nước trên đỉnh thác có nhiệt độ là $15^\circ C$, thì nhiệt độ cao nhất ở dưới chân thác là bao nhiêu? Giả sử toàn bộ năng lượng của thác khi chảy xuống đáy được sử dụng để tăng nhiệt độ của nước. Biết nhiệt dung riêng của nước là 4200 J/kg.K. Lấy $g=9,8\ m/s^2$.
	\choice
	{\True $16,9^\circ C$}
	{$20,1^\circ C$}
	{$23^\circ C$}
	{$15,0^\circ C$}}{\includegraphics[width=2.3cm]{images/5218}}
	\loigiai{
		\begin{itemize}
			\item Giả sử khối lượng của nước là $m (kg)$
			\item Giả sử toàn bộ năng lượng của thác khi chảy xuống đáy được sử dụng để tăng nhiệt độ của nước:
			\begin{align*}
				& m.g.h=m.c.\Delta t \\
				\Rightarrow \ & \Delta t=\dfrac{g h}{c}=\dfrac{9,8 . 807}{4200}=1,883^\circ C
			\end{align*}
			\item Nhiệt độ ở dưới chân thác là
			\begin{align*}
				t=15+1,883 \approx 16,9^\circ C
			\end{align*}
		\end{itemize}
	}
\end{ex}

\loai{Loại 4: Va chạm, áp dụng định lí biến thiên động năng}
\begin{ex} \nguon{3}
	\immini[thm]{Một viên đạn bằng đồng đang bay với vận tốc $200\  m/s$ thì va chạm vào một bức tường gỗ và nằm yên trong bức tường. Nhiệt dung riêng của đồng là $380\ J/kg . K$. Biết $90 \%$ động năng của đạn biến thành nội năng của viên đạn. Nhiệt độ của viên đạn sẽ tăng thêm bao nhiêu độ C (làm tròn kết quả  chữ số hàng phần mười)?
	\shortans[oly]{47,4 } }{\vspace*{0.7 cm}\includegraphics[width=8cm]{images/5387}}
	\loigiai{
		\begin{itemize}
			\item Trả lời ngắn: 47,4
			\item Động năng của viên đạn khi va chạm với tường: $W_{\text{đ}}=\dfrac{1}{2}mv^2$
			\item Theo bài ra:
			$$
			0,9W_{\text{đ}}=\Delta U=mc\Delta t \Rightarrow \Delta t=\dfrac{0,9W_{\text{đ}}}{mc}=\dfrac{0,9.\dfrac{1}{2}mv^2}{mc}=\dfrac{0,9.\dfrac{1}{2}v^2}{c}\Rightarrow \Delta t \approx 47,4^\circ C
			$$
		\end{itemize}
	}
\end{ex}
