\begin{ex}%book
Một người bơi dọc theo chiều dài $50\ m$ của bể bơi hết $20\ s$, rồi quay về lại chỗ xuất phát trong $22\ s$. Tính độ dịch chuyển, tốc độ trung bình và vận tốc trung bình.
\begin{itemize}
\item[a.] Trong lần đầu tiên theo chiều dài của bể.
\item[b.] Trong lần bơi về. 
\item[c.] Tổng suốt quãng đường đi và về.
\end{itemize}	
\haidong{12}
\loigiai{
Giả sử người này bơi từ A đến B rồi quay lại về A.\\
Chọn chiều dương từ A đến B.\\
Gốc toạ độ tại A.\\
Gốc thời gian là lúc bắt đầu xuất phát tại A.
\begin{itemize}
\item[a.] Trong lần bơi đầu tiên theo chiều dài của bể:
\begin{itemize}
\item Độ dịch chuyển: $\Delta d=d_2-d_1=50-0=50\ m.$
\item Tốc độ trung bình: $v_{tb}=\dfrac{s}{t}=\dfrac{50}{20}=2,5\ m/s$.
\item Vận tốc trung bình: $v=\dfrac{d_2-d_1}{t_2-t_1}=2,5\ m/s$
\end{itemize}	
\item[b.] Trong lần bơi về
\begin{itemize}
\item Độ dịch chuyển: $\Delta d=d_2-d_1=0-50=-50\ m.$
\item Tốc độ trung bình: $v_{tb}=\dfrac{s}{t}=\dfrac{50}{22}=2,27\ m/s$.
\item Vận tốc trung bình: $v=\dfrac{d_2-d_1}{t_2-t_1}=\dfrac{0-50}{22}=-2,27\ m/s$
\end{itemize}	
\item[c.] Trong suốt quãng đường đi và về: 
\begin{itemize}
\item Độ dịch chuyển: $\Delta d=d_2-d_1=0.$
\item Tốc độ trung bình: $v_{tb}=\dfrac{s}{t}=\dfrac{50+50}{20+22}=2,38\ m/s$.
\item Vận tốc trung bình: $v=\dfrac{d_2-d_1}{t_2-t_1}=0$.
\end{itemize}	
\end{itemize}	
}
\end{ex}

\item Nửa quãng đường đầu là $s_1$ và nửa quãng đường sau là $s_2$.

\item Thời gian đi trên nửa đường đầu và nửa đường sau là: $t_1=\dfrac{s}{v_1}$ và $t_2=\dfrac{s}{v_2}$.

\begin{align*}
v=\dfrac{s_1+s_2}{t_1+t_2}=\dfrac{2s}{\dfrac{s}{v_1}+\dfrac{s}{v_1}}=\dfrac{2v_1v_2}{v_1+v_2}=\dfrac{2.60.40}{60+40}=48\ km/h.
\end{align*}
\end{itemize}}
