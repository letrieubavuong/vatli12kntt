\loai{Loại 1: Bếp dầu}

\begin{ex}
	Khi dùng lò hiệu suất $H_1$ để làm chảy một lượng quặng, phải đốt hết $m_1$ kilôgam nhiên liệu có năng suất toả nhiệt $q_1$. Nếu dùng lò có hiệu suất $H_2$ để làm chảy lượng quặng trên, phải đốt hết
	$m_2=3m_1$ kilôgam nhiên liệu có năng suất toả nhiệt $q_2=0,5 q_1$. Khi đó
	\choice
	{$H_1=H_2$}
	{$H_1=2 H_2$}
	{$H_1=3 H_2$}
	{\True $H_1=1,5 H_2$}
	\loigiai{
		
		$H=\dfrac{Q_{q}}{Q}=\dfrac{Q_{q}}{q m} \Rightarrow \dfrac{H_1}{H_2}=\dfrac{q_2}{q_1} . \dfrac{m_2}{m_1}=0,5 . 3=1,5$.
		
	}
	\haidong{8}\end{ex}

\loai{Loại 2: Máy phát điện, động cơ điện}

\begin{ex}
	Người ta dùng một máy hơi nước hiệu suất $10 \%$ để đưa $720\ m^3$ nước lên độ cao $9\ m$. Biết năng suất tỏa nhiệt của than đá là $27 . 10^{6}\ J/kg$. Lấy $g=10\ m/s^2$. Lượng than đá tiêu thụ là
	\choice
	{$22\ kg$}
	{$23\ kg$}
	{\True $24\ kg$}
	{$25\ kg$}\loigiai{
		
		$A=m g h=V \rho g h=720 . 10^3 . 10 . 9=6,48 . 10^{7}\ J$
		
		$Q=\dfrac{A}{H}=\dfrac{6,48 . 10^{7}}{0,1}=64,8 . 10^{7}\ J$
		
		$m=\dfrac{Q}{q}=\dfrac{64,8 . 10^{7}}{27 . 10^{6}}=24\ kg$.
		
	}
	\haidong{4}\end{ex}

\loai{Loại 3: Động cơ ô tô, xe máy}

\begin{ex} \nguon{1}
	Động cơ của một máy bay có công suất $2.10^{6} \ W$ và hiệu suất $30 \%$. Biết năng suất toả nhiệt của xăng là $4,6 . 10^{7} \ J / kg$. Với một tấn xăng, máy bay có thể bay được trong
	\choice
	{0,92 giờ}
	{3,92 giờ}
	{\True 1,92 giờ}
	{2,92 giờ}
	\loigiai{
		\begin{itemize}
			\item 1 tấn $=1000 \ kg$
			
			\item Năng lượng do một tấn xăng toả ra: $Q=mq=1000 . 4,6 . 10^{7}=4,6 . 10^{10} \ J$
			\item Ta có: $H=\dfrac{A}{Q}$
			
			\item  Công do máy bay động cơ sinh ra: $A=H . Q=0,30 . 4,6 . 10^{10}=1,38 . 10^{10} \ J$
			\item  Thời gian máy bay bay là: $t=\dfrac{A}{\mathcal{P}}=\dfrac{1,38 . 10^{10}}{2.10^{6}}=6900=1,92\ h$
		\end{itemize}
	}
\end{ex}

\begin{ex} \nguon{1}
	Một ô tô có công suất $5000 \ W$ chuyển động với vận tốc $72 \ km / h$ chạy quãng đường $450 \ km$ thì động cơ tiêu thụ 9 lít xăng. Biết năng suất toả nhiệt của xăng là $4,6.10^{7} \ J / kg$, khối lượng riêng của xăng là $700 \ kg / m^3$. Hiệu suất của động cơ ô tô là bao nhiêu \% (làm tròn kết quả  đến chữ số hàng phần mười)?
	\shortans[oly]{ 38,8} 
	\loigiai{
		\begin{itemize}
			\item TLN: 38,8
			\item 9 lít $=9.10^{-3} \ m^3$
			\item Khối lượng của 9 lít xăng là: $m=\rho_{\text {xăng}} V=700 . 9 . 10^{-3}=6,3 \ kg$
			\item Năng lượng do 9 lít xăng bị đốt tỏa ra là: $Q=m q=6,3 . 4,6 . 10^{7}=2,898 . 10^{8} \ J$
			\item Thời gian ô tô chạy là: $t=\dfrac{s}{v}=\dfrac{450}{72}\ h=22500\ s$
			\item  Công do động cơ ô tô sinh ra: $A=\mathcal{P}. t=1,125.10^{8}\ J$
			\item Hiệu suất của động cơ ô tô: $H=\dfrac{A}{Q} . 100 \%=38,82 \%$
		\end{itemize}
	}
\end{ex}

\begin{ex}
	Một động cơ xe máy khi chạy với tốc độ $25 \ km / h$ thì sẽ tiêu thụ hết 1,7 lít xăng cho mỗi $100 \ km$. Công suất trung bình của động cơ là $\mathcal{P}$. Cho biết hiệu suất của động cơ là $20 \%$ và năng suất tỏa nhiệt của xăng là $46 . 10^{6} \ J /kg$. Cho biết khối lượng riêng của xăng là $700 \ kg / m^3$. Giá trị của $\mathcal{P}$ là bao nhiêu $kW$ (làm tròn kết quả  đến chữ số hàng phần trăm)?
	\shortans[oly]{ 0,76}
	\loigiai{
		\begin{itemize}
			\item TLN: 0,76
			\item Khối lượng của 1,7 lít xăng: $m=V . \rho =1,7 . 10^{-3} . 700=1,19\ kg$
			\item Nhiệt lượng khi đốt cháy hết 1,7 lít xăng: $Q_1=q . m=46 . 10^{6} . 1,19=54,74 . 10^{6}\ J$
			\item Vì hiệu suất của động cơ là $20 \%$ nên: $A=H . Q_1=0,2 . 54,74 . 10^{6}=10,948 . 10^{6}\ J$
			\item Thời gian đi được quãng đường 100\ km: $t=\dfrac{s}{v}=\dfrac{100}{25}=4 \ h=14400\ s$
			\item Công suất trung bình của động cơ: $\mathcal{P}=\dfrac{A}{t}=\dfrac{10,948 . 10^{6}}{14400} \approx 760,27 \ W=0,76\ kW$
		\end{itemize}
	}
	\haidong{8}\end{ex}

	}
