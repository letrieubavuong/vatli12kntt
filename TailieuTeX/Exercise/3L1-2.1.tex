	{\True với mỗi áp suất xác định, mỗi chất rắn kết tinh nóng chảy và đông đặc ở cùng một nhiệt độ xác định}

		\begin{itemize}
			\item Phát biểu đúng là: \emph{Mỗi chất rắn kết tinh nóng chảy và đông đặc ở cùng một nhiệt độ xác định trong điều kiện áp suất xác định}.
			
			\item Giải thích chi tiết:
			\begin{itemize}
				\item \textbf{Mỗi chất rắn nóng chảy ở một nhiệt độ xác định, không phụ thuộc vào áp suất bên ngoài}: Sai. Nhiệt độ nóng chảy phụ thuộc vào áp suất bên ngoài.
				\item \textbf{Nhiệt độ đông đặc của chất rắn kết tinh không phụ thuộc vào áp suất bên ngoài}: Sai. Nhiệt độ đông đặc cũng phụ thuộc vào áp suất bên ngoài.
				\item \textbf{Mỗi chất rắn kết tinh nóng chảy và đông đặc ở cùng một nhiệt độ xác định trong điều kiện áp suất xác định}: Đúng. Trong điều kiện áp suất xác định, điểm nóng chảy và đông đặc của một chất rắn kết tinh là như nhau.
				\item \textbf{Mỗi chất rắn nóng chảy ở nhiệt độ nào thì cũng sẽ đông đặc ở nhiệt độ đó}: Cần làm rõ điều kiện áp suất xác định, vì phát biểu này có thể lập lờ.
		\end{itemize}\end{itemize}
	}
\end{ex}

				\item \textbf{Sự đông đặc}: Sai. Đông đặc của chất rắn kết tinh cũng xảy ra ở nhiệt độ xác định.
				\item \textbf{Sự sôi}: Sai. Sự sôi xảy ra ở nhiệt độ xác định gọi là nhiệt độ sôi.
				\item \textbf{Sự bay hơi}: Đúng. Bay hơi có thể xảy ra ở bất kì nhiệt độ nào dưới điểm sôi, không có nhiệt độ cố định.
		\end{itemize}\end{itemize}
	}
