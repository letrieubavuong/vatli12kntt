\begin{vdc}
Hành khách này tham gia mấy chuyển động?
\end{vdc}
\begin{vdc}
Làm cách nào để xác định được vận tốc của hành khách đối với mặt đường?
\end{vdc}
\begin{vdc}
Nếu người này chuyển động về cuối đoàn tàu với vận tốc có cùng độ lớn $1\ m/s$ thì vận tốc của hành khách đối với mặt đường là bao nhiêu $m/s$?
\shortans[oly]{9}
\end{vdc}
\haidong{20}
\loigiai{
\begin{itemize}
\item[a)] Hành khách này tham gia hai chuyển động: Chuyển động với vận tốc $1\ m/s$ so với sàn tàu và chuyển động do tàu kéo đi (chuyển động kéo theo) với vận tốc bằng vận tốc của tàu so với mặt đường. 
\item Chuyển động của hành khách so với mặt đường là tổng hợp của hai chuyển động trên.
\item[b)] Nếu gọi $\vec v_{1,2}$ là vận tốc của hành khách so với tàu, 
$\vec v_{2,3}$ là vận tốc của tàu so với mặt đường và 
$\vec v_{1,3}$ là vận tốc của hành khách so với mặt đường thì: 
$\vec v_{1,3}=\vec v_{1,2}+\vec v_{2,3}$.
\item Vì các chuyển động trên đều là chuyển động thẳng theo hướng chạy của đoàn tàu nên: $v_{1,3}=v_{1,2}+v_{2,3}=1\ m/s + 10\ m/s=11\ m/s$.
\item Hướng của vận tốc là hướng đoàn tàu chạy.
\end{itemize}
\item[c)] 		
\begin{itemize}
\item Đổi: $36\ km/h = 10\ m/s$
\item Gọi:
\item[-] $\vec v_{1,2}$ là vận tốc của hành khách so với tàu
\item[-]$\vec v_{2,3}$ là vận tốc của tàu so với mặt đường
\item[-]$\vec v_{1,3}$ là vận tốc của hành khách so với mặt đường
\item Suy ra, ta có: $\vec v_{1,3}=\vec v_{1,2}+\vec v_{2,3}$
\item Do hành khách chuyển động về cuối đoàn tàu, tức là ngược chiều chuyển động của đoàn tàu nên ta có: $v_{1,3}=-v_{1,2}+v_{2,3}=-1+10=9\ m/s$.
\item Vậy vận tốc của hành khách đối với mặt đường trong trường hợp này là 9 m/s.
\end{itemize}
}
\end{vidu}

\begin{vdc}
Người soát vé đứng yên trên tàu.

\end{itemize}
}

\end{vdc}
\begin{vdc}
Khoảng cách từ bến tàu $A$ đến bến tàu $B$ là bao nhiêu $m/s$ (làm tròn kết quả đến chữ số hàng phần mười)?
\shortans[oly]{7,2}
\end{vdc}
\loigiai{
Gọi:
\begin{itemize}
\item[-] $\vec v_{1,2}$ là vận tốc của canô so với nước
\item[-] $\vec v_{2,3}$ là vận tốc của nước so với bờ
\item[-] $\vec v_{1,3}$ là vận tốc của canô so với bờ
\item Ta có: $\vec v_{1,3}=\vec v_{1,2}+\vec v_{2,3}$
\item Khi canô chạy trên mặt nước yên lặng, tức $v_{2,3}=0$, ta có: $v_{1,2}=v_{1,3}=21,5\ km/h$.

\item Khi canô quay lại, ta có: $v'_{1,3}=v_{1,2}-v_{2,3}=21,5-v_{2,3}\Rightarrow t_1=\dfrac{d}{21,5-v_{2,3}}\Leftrightarrow 2=\dfrac{d}{21,5-v_{2,3}}\Leftrightarrow 43=d+2.v_{2,3}$\quad (2)
\item Từ (1) và (2) ta suy ra: $\begin{cases}
d=28,67\ km\\
v_{2,3}=7,17\ km/h
\end{cases}$
\item Vậy vận tốc chảy của dòng sông là 7,17 km/h.
\end{itemize}
}
\end{vidu}

\item Vận tốc của thuyền $(1)$ đối với nước $(2)$ là $\vec{v}_{12} \Rightarrow  v_{12} = 14\ km/h$.
\item Vận tốc của nước $(2)$ đối với bờ (3) là $\vec{v}_{23} \Rightarrow  v_{23} = 9
\ km/h$.
\item Theo công thức cộng vận tốc ta có: $\vec{v}_{13}=\vec{v}_{12}+\vec{v}_{23}\quad (*)$
\item Theo đề bài vận tốc của thuyền đối với dòng nước là $14\ km/h > 0 \Rightarrow$  chiều dương được chọn theo chiều của $\vec{v}_{12}$.

\end{itemize}

$v_{13} = 5\ km/h > 0 \Rightarrow  \vec{v}_{13}$ cùng chiều dương.
\end{note}

\begin{itemize}
\item Thuyền chuyển động ngược chiều với nước nên vận tốc của thuyền đối với bờ là:
\begin{align*}
v_{\ct{thuyền-bờ}} = v_{\ct{ngược}} = v_{\ct{thuyền-nước}} - v_{\ct{nước-bờ}} = 14 - 9 = 5\ km/h.
\end{align*}
\end{itemize}}
