Khi đó quỹ đạo proton và $\alpha$ là các đường tròn có bán kính lần lượt là $R_P = 30$ cm và $R_\alpha$. Biết khối lượng $m_p = 1,672. 10^{-27}$ kg, $m_\alpha = 6,688. 10^{-27}$ kg. Bán kính quỹ đạo của hạt nhân helium $R_\alpha$ là
	\choice
	{$R_\alpha = 21,2$ cm}
	{\True $R_\alpha = 42,4$ cm}
	{$R_\alpha = 30$ cm}
	{$R_\alpha = 15$ cm}
	\loigiai{\begin{itemize}
			\item Động năng của hạt sau khi được tăng tốc bởi U là: 
			\begin{align*}
			&\dfrac{1}{2}m{{v}^{2}}=\left| q \right|U \Rightarrow v=\sqrt{\dfrac{2qU}{m}}\\
			\Rightarrow\ &R=\dfrac{mv}{\left| q \right|B}=\dfrac{\sqrt{2mU}}{B\sqrt{\left| q \right|}}.
			\end{align*}
			\item Điện tích hạt nhân helium là $q_\alpha = 2e$, điện tích hạt proton là 
			\begin{align*}
				q_p = e.
			\Rightarrow \dfrac{R_{\alpha}}{R_p}=\sqrt{\dfrac{m_{\alpha}}{m_p}. \dfrac{q_p}{q_{\alpha}}}=\sqrt{2}\Rightarrow R_{\alpha}=30\sqrt{2} cm.
			\end{align*}
	\end{itemize}}
