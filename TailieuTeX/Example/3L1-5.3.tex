			\item Ta viết các phương trình cân bằng nhiệt:
			\begin{itemize}
				\item Bình 1: $\left(q_{X}+m_0 c_0\right) \Delta t_1=m c \Delta t_2$ \quad \text {(1)}
				\item Bình 2: $q_{X} \Delta t=N m c \Delta t$ \quad \text {(2)}
			\end{itemize}
			\item Thể tích của lượng nước bằng thể tích $(N-1)$ mẩu hợp kim:
			\begin{align*}
				V_0=(N-1) V_{m} \Rightarrow \dfrac{m_0}{\rho_0}=(N-1) \dfrac{m}{\rho} \quad \text {(3)}
			\end{align*}
			Từ (2) $\Rightarrow q_{X}=N m c$\\
			Thế vào (1) $\Rightarrow\left(N m c+m_0 c_0\right) \Delta t_1=m c \Delta t_2$
			$\Rightarrow m_0 c_0 \Delta t_1=m c\left(\Delta t_2-N \Delta t_1\right)$ $\Rightarrow c=\dfrac{m_0}{m} \dfrac{c_0 \Delta t_1}{\Delta t_2-N \Delta t_1}$\\
			Kết hợp với (3) $\Rightarrow c=\dfrac{(N-1) \rho_0}{\rho} . \dfrac{c_0 \Delta t_1}{\Delta t_2-N \Delta t_1}=800 \ J/kg . K$.
		\end{itemize}
	}
