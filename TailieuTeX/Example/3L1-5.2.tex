\loai{Loại 1: Tính Q của 1 vật}
\begin{vidu}
	Biết nhiệt dung riêng của nước là 4180 J/kg.K. Nhiệt lượng cần cung cấp để đun nóng 15 kg nước từ $25^\circ C$ đến $60^\circ C$ xấp xỉ là
	\choice
	{$1,2.10^6\ J$}
	{\True $2,2.10^6\ J$}
	{$1,1.10^6\ J$}
	{$2,2.10^4\ J$}
	\loigiai{
		\begin{itemize}
			\item Dùng công thức: $Q=mc\Delta t$
			\item Ta có: $\Delta t=(273+60)-(273+25)=35\ K$
			\item $Q=15.4180.35=2,2.10^6\ J$
		\end{itemize}
	}
	\haidong{3}\end{vidu}

\loai{Loại 2: Tính Q của 1 vật - bài toán ngược}
\begin{vidu}%[1P6-1.2K][Loai2] 
	Nhiệt lượng một vật đồng chất thu vào là 6900 J làm nhiệt độ vật tăng thêm 50\doC. Khối lượng của vật là 300 g. Nhiệt dung riêng của chất làm vật là
	\choice
	{\True 460 J/kg.K}
	{1150 J/kg.K}
	{8100 J/kg.K}
	{41,4 J/kg.K}
	\loigiai{
		Áp dụng công thức: $Q=m.c.(t_2-t_1)\Rightarrow c=\dfrac{Q}{m.(t_2-t_1)}=460\ J/kg.K.$
	}
	\haidong{6}\end{vidu}

\loai{Loại 3: Tính Q theo P của 1 vật}
\begin{vidu}
	Có 20 lít nước ở nhiệt độ $20^\circ C$. Cho khối lượng riêng của nước là $1000\ kg / m^3$ và nhiệt dung riêng của nước là $4186\ J/kg.K$. Dùng một thiết bị điện có công suất $2,5\ kW$ để đun lượng nước trên lên tới $70^\circ C$ thì thời gian truyền nhiệt lượng cần thiết là bao nhiêu giây? Biết chỉ có $80 \%$ điện năng tiêu thụ được dùng để làm nóng nước.
	\shortans[oly]{2093}
	\loigiai{
		\begin{itemize}
			\item Khối lượng của 20 lít nước: $m=V . \rho =20\ kg$
			\item Nhiệt lượng nước thu vào: $Q=m . c . \Delta t=20 . 4186 .(70-20)=4186 . 10^3\ J$
			\item Vì hiệu suất là $80 \%$ nên: $A=\dfrac{Q}{H}=\dfrac{4186 . 10^3}{0,8}=5232,5 . 10^3\ J$
			\item Thời gian $t=\dfrac{A}{P}=2093\ s$
		\end{itemize}
		
		\textbf{\textbf{Trả lời ngắn:} }2093
		
	}
	\haidong{8}\end{vidu}

\loai{Loại 6: Tính Q của hệ 2 vật - bài toán ngược}
\begin{vidu}%[1P6-1.2G][Loai10]
	Một ấm đun nước bằng nhôm có có khối lượng 400 g, chứa 3 lít nước được đun trên bếp. Khi nhận được nhiệt lượng 740 kJ thì ấm đạt đến nhiệt độ $80^{\circ}C$.  Bỏ qua sự trao đổi nhiệt với môi trường bên ngoài. Cho khối lượng riêng của nước là $1000\ kg / m^3$, nhiệt dung riêng của nhôm và nước lần lượt là  $c_{Al}=880$ J/kg.K, $c_{H_2O}=4190$ J/kg.K. Nhiệt độ ban đầu của ấm là 
	\choice
	{8,15$^{\circ}C$}
	{8,15 K}
	{\True 22,7$^{\circ}C$}
	{22,7 K}
	\loigiai{ 
		\Binhluan{
			\begin{itemize}
				\item Nhiệt lượng trao đổi của nước: 			
				\begin{align*}
					Q_{H_2O}=m_{H_2O}.c_{H_2O}(t-t_1)=1005600-12570t_1.
				\end{align*}
				\item Nhiệt lượng trao đôi của ấm nhôm: 
				\begin{align*}
					Q_{Al}=m_{Al}.c_{Al}(t-t_1)=28160-352t_1.
				\end{align*}
				\item Vì ấm và nước nhận được nhiệt lượng 740 kJ, ta có:
				\begin{align*}
					Q_{H_2O}+Q_{Al}=740.10^3\Rightarrow t_1=22,7^{\circ}C.
				\end{align*}
			\end{itemize}
		}
		{Đây không phải là quá trình cân bằng nhiệt}
	}
	\haidong{7}\end{vidu}
\loai{Loại 7: Tính Q theo P của hệ 2 vật}
\begin{vidu}
	Một ấm đồng khối lượng 300 g chứa 1 lít nước ở nhiệt độ $15^\circ C$. Biết trung bình mỗi giây bếp truyền cho ấm một nhiệt lượng là $500\ J$. Bỏ qua sự hao phí về nhiệt ra môi trường xung quanh. Lấy nhiệt dung riêng của đồng là $380 \ J /kg.K$ và của nước là $4186\ J /kg.K$, khối lượng riêng của nước là $1000\ kg / m^3$. Thời gian để đun sôi nước trong ấm là
	\choice
	{\True 731 giây}
	{365 giây}
	{73,1 giây}
	{500 giây}
	\loigiai{
	Nhiệt lượng cần thiết để đun nóng nước và ấm đồng:
	\[ Q = m_{\text{Cu}} c_{\text{Cu}} \Delta T + m_{\text{H}_2\text{O}} c_{\text{H}_2\text{O}} \Delta T = 0.3 \times 380 \times 85 + 1 \times 4186 \times 85 = 365500 \ \text{J} \]
	Thời gian đun sôi:
	\[ t = \frac{Q}{P} = \frac{365500}{500} = 731 \ \text{s} \]
	Vậy thời gian cần thiết để nước trong ấm bắt đầu sôi là \(731 \ \text{s}\).
	
	}
	\haidong{6}
\end{vidu}

Biết rằng cường độ bức xạ năng lượng Mặt Trời thu được trên Trái Đất vào buổi trưa là khoảng $1000 \ W / m^2$. Lấy nhiệt dung riêng của nước là $4180 \ J/kg$.K. Độ tăng nhiệt độ của 150 kg nước khi máy trên hoạt động trong 2 giờ vào buổi trưa là bao nhiêu Kelvin (làm tròn kết quả  đến chữ số hàng đơn vị)?

	\shortans[oly]{55 }
	\loigiai{
		Công suất hoạt động của 2 tấm thu năng lượng là: 
		
		$P=IS=1000.2.2.2=5000 \ W$ 
		
		$Q=$ H.P. $\Delta t=0,96.5000.7200=3456.10^{4} \ J$

		$$
		\Delta t=\dfrac{Q}{mc}=\dfrac{3456.10^{4}}{150.4180} \approx 55^\circ C
		$$
