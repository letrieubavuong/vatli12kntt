\begin{align*}
\dfrac{120}{961}K_{\alpha}+\dfrac{4}{961}K_{\alpha}=K_{\alpha}+W\Leftrightarrow -\dfrac{27}{31}K_{\alpha}=W=-2,70\Rightarrow K_{\alpha}=3,10\ MeV.
