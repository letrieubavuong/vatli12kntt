
\begin{vidu}%[2P5-4.2K][Loai1]
Một ống dây dài 50 cm, diện tích tiết diện ngang của ống là 10 $cm^2$ gồm 1000 vòng dây. Hệ số tự cảm của ống dây là
\choice
{$0,251$ H}
{$6,{28.10^{- 2}}$ H}
{$2,{51.10^{- 2}}$ mH}
{\True $2,51$ mH}
\loigiai
{Áp dụng công thức: $L = 4\pi .{10^{- 7}}.\dfrac{N^2}{\ell}.V=2,51\ mH$}
\end{vidu}
\begin{vidu}%[2P5-4.2K][Loai1]
Một ống dây có độ tự cảm L; ống dây thứ hai có số vòng dây tăng gấp đôi, và diện tích mỗi vòng dây giảm một nửa so với ống dây thứ nhất. Nếu hai ống dây có chiều dài như nhau thì độ tự cảm của ống dây thứ hai là
\choice
{L}
{\True 2L}
{$\dfrac{L}{2}$}
{4L}
\loigiai{
\begin{itemize}
\item Ta có, hệ số tự cảm mỗi ống dây là: ${L_1} = 4\pi {.10^{- 7}}.\dfrac{N_1^2}{\ell}.S_1;\ {L_2} = 4\pi {.10^{- 7}}.\dfrac{N_2^2}{\ell}.S_2$
\item Lập tỷ số ta được: $\dfrac{L_1}{L_2} = \dfrac{4\pi {.10^{- 7}}.\tfrac{N_1^2}{\ell}.S_1}{4\pi {.10^{- 7}}.\tfrac{N_2^2}{\ell}.S_2}  = {\left( {\dfrac{N_1}{N_2}} \right)^2}\dfrac{S_1}{S_2} = \dfrac{1}{2}\Rightarrow L_2=2L_1=2L.$
\end{itemize}
}
\end{vidu}
\begin{vidu}%[2P5-4.2K][Loai1]
Một ống dây dài $\ell = 40$ cm gồm $N = 800$ vòng có đường kính mỗi vòng 10 cm. Tìm hệ số tự cảm của ống dây. Lấy $\pi^2 = 10$.
\choice
{$L = 12$ mH}
{$L = 26$ mH}
{\True $L = 16$ mH}
{$L = 20$ mH}
\loigiai{
\begin{itemize}
\item Hệ số tự cảm của ống dây:
\begin{align*}
L = 4\pi.10^{-7}.\dfrac{N^2}{\ell}.S = 4\pi .10^{-7}.\dfrac{N^2}{\ell}.\dfrac{\pi D^2}{4}= 160.10^{-4}\ H = 16\ mH.
\end{align*}
\end{itemize}
}
\end{vidu}
\begin{vidu}%[2P5-4.2K][Loai1]
Một ống dây hình trụ có đường kính 16 cm và chiều dài 80 cm. Một sợi dây có chiều dài 100 m được quấn một cách đều đặn theo chiều dài của ống. Hệ số tự cảm của cuộn dây là
\choice
{$2,13. 10^{-3}$ H}
{$3,15. 10^{-3}$ H}
{$7,14. 10^{-3}$ H}
{\True $1,25. 10^{-3}$ H}
\loigiai{
\begin{itemize}
\item Mỗi vòng dây quấn có chu vi là $P = \pi d = 16\pi$  cm. 
\item Số vòng quấn trên ống là $N=\dfrac{L}{P}=  199 $ vòng.
\item Tiết diện ống $S = \pi R^2 = 0,02\ m^2 \Rightarrow L=4\pi {{.10}^{-7}}\dfrac{{{N}^{2}}}{\ell}S=1,25. 10^{-3}$ H.
\end{itemize}
}
\end{vidu}
