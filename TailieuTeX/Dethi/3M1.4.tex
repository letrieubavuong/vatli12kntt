\choice
{$50^\circ C$}
{$0^\circ C$}
{$100^\circ C$}
{\True $273^\circ C$}
\loigiai{
\begin{itemize}
\item Gọi $t$ là nhiệt độ Celsius cần tìm. Nhiệt độ tuyệt đối tương ứng là $T = t + 273$.

\item Giải phương trình, ta được $t = 273^\circ C$.
\end{itemize}
}
