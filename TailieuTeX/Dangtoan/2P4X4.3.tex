\begin{dang}{Dây dẫn cân bằng dưới tác dụng của lực từ, trọng lực, lực ma sát và lực căng}
\Anlythuyet{\begin{description}
	\item[Bước 1:] Phân tích tất cả các lực tác dụng lên đoạn dây dẫn với
	\item[Bước 2:] Vẽ hình sao cho các lực đó cân bằng lẫn nhau. Sau đó tìm các lực theo yêu cầu.
	\end{description}}
\end{dang}
