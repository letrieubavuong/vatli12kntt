\begin{dang*}{Khung dây quay trong từ trường}
\Anlythuyet{\begin{itemize}
	\item Từ thông nhanh pha hơn suất điện động một góc $\dfrac{\pi}{2}$
	\item Vì e và $\Phi$ vuông pha nên ta có công thức độc lập với thời gian:
	\begin{align*} 
		\boder{\left( \dfrac{e}{E_0} \right)^2+\left( \dfrac{\Phi}{\Phi_0} \right)^2=1 \Rightarrow \Phi _{0}^2=\Phi^2+\dfrac{e^2}{\omega^2}}
	\end{align*}
\item Suất điện động hiệu dụng: $\boder{E=\dfrac{E_0}{\sqrt2}}$
\end{itemize}}
\end{dang*}
