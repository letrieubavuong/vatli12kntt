\begin{dang}{Biến thiên nội năng}
\begin{itemize}
\item Độ biến thiên thế năng: $\Delta W_t=mgh_2-mgh_1$
\item Độ biến thiên động năng: $\Delta W_{\text{đ}}=\dfrac{1}{2}mv_2^2-\dfrac{1}{2}mv_1^2$
\item Nếu toàn bộ thế năng hoặc động năng chuyển hoá thành nội năng thì: $\Delta U=\Delta W_t$ hoặc $\Delta U=\Delta W_\text{đ}$
\end{itemize}
\end{dang}
