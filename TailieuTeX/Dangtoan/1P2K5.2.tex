\begin{dang}{CHUYỂN ĐỘNG THẲNG BIẾN ĐỔI ĐỀU}
\paragraph{Sử dụng công thức trong chuyển động thẳng biến đổi đều}
\begin{itemize}
\item Khi dùng công thức thì ta quy ước chọn $v_0>0$ cho đơn giản. Khi đó
\begin{itemize}
\item Khi vật chuyển động nhanh dần đều thì $a>0$.
\item Khi vật chuyển động chậm dần đều thì $a<0$.
\end{itemize}
\item Quãng đường đi được trong giây thứ n
\begin{center}

\begin{center}
\begin{tabular}{|c|c|c|c|}
\hline

& n \textit{giây}	& ($n-1$) \textit{giây} & \textit{giây thứ n} \\ 
\hline 
\textbf{Công thức} & 
$s_n=v_0.n+\dfrac{1}{2}an^2$	& $s_{n-1}=v_0.\left(n-1\right)+\dfrac{1}{2}a.\left(n-1\right)^2$   & $\Delta s=s_n-s_{n-1}$ \\ 
\hline 

\end{tabular}

\end{itemize}
\paragraph{Phương trình tọa độ của chuyển động thẳng biến đổi đều}
\begin{align*}\boder{x=x_0+v_0t+\dfrac{1}{2}at^2}\end{align*} 
\begin{itemize}
\item[$\bullet$] $x_0$ là toạ độ tại thời điểm ban đầu $t_0=0$.
\item[$\bullet$] t là thời gian chuyển động.
\end{itemize}
