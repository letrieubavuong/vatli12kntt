\begin{dang*}{Lí thuyết}
\Anlythuyet{
\begin{itemize}
\item Phản ứng nhiệt hạch hiện nay chưa kiểm soát được.
\item Một phản ứng nhiệt hạch toả năng lượng nhỏ hơn một phản ứng phân hạch nhưng xét theo khối lượng nhiên liệu thì $1\ g$ chất nhiệt hạch toả năng lượng lớn hơn $1\ g$ chất phân hạch.
\end{itemize}
}
\end{dang*}
