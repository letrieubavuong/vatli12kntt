\begin{dang*}{Lí thuyết bước sóng và thông tin vô tuyến}
\Anlythuyet{
\begin{itemize}
\item Trong máy thu có thể sử dụng đồng thời hai mạch khuếch đại cao tần và khuếch đại âm tần. Khuếch đại cao tần trước khi tách sóng rồi tới khuếch đại âm tần.
\item Mạch chọn sóng hoạt động dựa trên hiện tượng cảm ứng điện từ và cộng hưởng điện.
\item Truyền hình mặt đất dùng sóng ngắn, truyền hình vệ tinh dùng các sóng cực ngắn.
\item Muốn truyền thông tin đi xa, người ta phải đặt các đài tiếp sóng trung gian hoặc dùng vệ tinh nhân tạo để thu rồi phát trở về Trái Đất.
\end{itemize}
}
\end{dang*}
