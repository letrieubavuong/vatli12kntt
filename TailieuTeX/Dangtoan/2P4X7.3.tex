\begin{dang}{Chu kì, tần số, bán kính quỹ đạo của hạt chuyển động vuông góc với từ trường}
\Anlythuyet{
Khi điện tích chuyển động theo phương vuông góc với đường sức từ trường đều thì chuyển động của điện tích đó là chuyển động tròn đều với lực Lorentz đóng vai trò lực hướng tâm.
\begin{align*}
	f_L=F_{ht} \Leftrightarrow |q|vB=\dfrac{mv^2}{R} \Rightarrow 
	\begin{cases}
		\text{Vận tốc}\ v = \dfrac{|q|BR}{m}\\
		\text{Tốc độ góc}\ \omega=\dfrac{v}{R}=\dfrac{|q|B}{m}\\
		\text{Chu kì}\ T=\dfrac{2\pi}{\omega}=2\pi\dfrac{m}{|q|B}\\
		\text{Tần số}\ f=\dfrac{1}{T}=\dfrac{1}{2\pi}\dfrac{|q|B}{m}
	\end{cases}
\end{align*}
}
\end{dang}
