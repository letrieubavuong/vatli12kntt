\begin{dang*}{Lí thuyết phương trình trạng thái}
	\Anlythuyet{\begin{itemize}
			\item Phương trình trạng thái chỉ áp dụng cho một lượng khí lí tưởng xác định.
			\item Đồ thị biến đổi trạng thái thường là tổng hợp các đẳng quá trình nên ta vận dụng đồ thị các đẳng quá trình để suy luận.
	\end{itemize}}
\end{dang*}
