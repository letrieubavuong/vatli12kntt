\begin{dang*}{Lí thuyết}
\Anlythuyet{
\begin{itemize}
	\item Xung quanh dòng điện không đổi có từ trường.
	\item Khi truyền từ môi trường này sang môi trường khác tần số sóng điện từ không đổi.
	\item Quanh điện tích đứng yên chỉ tồn tại điện trường tĩnh.
	\item Từ trường không tác dụng lực lên điện tích đứng yên.
\end{itemize}
}
\end{dang*}
