\begin{dang}{Hiệu suất - Điện năng tiêu thụ của động cơ điện}
\Anlythuyet{		\begin{itemize}
		\item \textbf{Hiệu suất} của động cơ: \boder{H = \dfrac{\mathcal{P}_{ci}}{\mathcal{P}}}
		\item Phần trăm hao phí: \hspace{0.6cm} \boder{h=\dfrac{\mathcal{P}_{hp}}{\mathcal{P}}=1-H}
		\item Điện năng tiêu thụ: \hspace{0.6cm}  \boder{A = \mathcal{P}t} 
		\item Năng lượng cơ có ích: \hspace{0.37cm}	\boder{A_{ci} = \mathcal{P}_{ci}.t}
	\end{itemize}
	\begin{note}
		Công suất tiêu thụ của động cơ gồm hai phần: công suất cơ học và công suất hao phí do tỏa nhiệt.
		\begin{align*}
			UI\cos\varphi=I^2r+\mathcal{P}_{ci}
		\end{align*}
		\end{note}}
\end{dang}
