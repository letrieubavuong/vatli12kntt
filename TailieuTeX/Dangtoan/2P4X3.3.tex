\begin{dang}{Từ trường do nhiều dòng điện tròn và dài gây ra}
\Anlythuyet{\begin{description}
\item[Bước 1:] Xác định vector cảm ứng từ của từng dòng điện thẳng và tròn gây ra tại điểm đang xét.
\item[Bước 2:] Áp dụng nguyên lý chồng chất từ trường để xác định vector cảm ứng từ tổng hợp tại điểm đang xét
\begin{align*}
\boder{\vec{B}=\vec{B}_1+\vec{B}_2+\cdots+\vec{B}_n}
\end{align*}
\end{description}}
\end{dang}
