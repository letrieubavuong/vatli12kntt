\begin{dang}{Phương trình phóng xạ. Chu kì, hằng số phóng xạ}
\Anlythuyet{
\begin{itemize}
\item Giả sử hạt nhân mẹ $\Hn{{A_1}}{{Z_1}}{X}$ tự động phát phân rã tạo thành hạt nhân con $\Hn{{A_2}}{{Z_2}}{Y}$ và tia phóng xạ $\Hn{{A_3}}{{Z_3}}{Z}$.\\
Ta có phương trình phóng xạ: $\Hn{{A_1}}{{Z_1}}{X}\to \Hn{{A_2}}{{Z_2}}{Y} + \Hn{{A_3}}{{Z_3}}{Z}$, với
$\left\{\begin{aligned}& A_1=A_2+A_3 \\
& Z_1=Z_2+Z_3 
\end{aligned}\right.$
\end{itemize}
}
\end{dang}
