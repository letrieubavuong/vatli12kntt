\begin{dang*}{Lí thuyết}
\Anlythuyet{\begin{itemize}
\item Khi không khí lạnh đi, tới nhiệt độ nào đó hơi nước trong không khí trở nên bão hoà, không khí có độ ẩm cực đại  
\item Độ ẩm cực đại là độ ẩm tuyệt đối lớn nhất ở một nhiệt độ
\item Khi nhiệt độ tăng độ ẩm tuyệt đối và độ ẩm cực đại đều tăng nhưng độ ẩm tuyệt đối tăng chậm hơn độ ẩm cực đại nên độ ẩm tỉ đối giảm
\end{itemize}}
\end{dang*}
