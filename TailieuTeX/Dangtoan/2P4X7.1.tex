\begin{dang*}{Lí thuyết về lực Lorentz}
\Anlythuyet{\begin{itemize}
	\item Điện tích chuyển động trong từ trường đều thì lực Lorentz không nằm trong mặt phẳng chứa vector vận tốc của hạt.
	\item Trong từ trường đều có hai điện tích trái dấu chuyển động cùng chiều. Lực Lorentz tác dụng lên các điện tích đó sẽ ngược hướng.
\end{itemize}}
\end{dang*}
