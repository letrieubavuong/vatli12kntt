\begin{dang*}{Lí thuyết}
\Anlythuyet{\begin{itemize}
	\item Ampe kế đo cường độ dòng điện xoay chiều và vôn kế đo điện áp xoay chiều dựa vào tác dụng nhiệt  của dòng điện nên gọi là ampe kế nhiệt và vôn kế nhiệt.\\
	Số chỉ của chúng là cường độ hiệu dụng và điện áp hiệu dụng của dòng điện xoay chiều.
	\item Không thể sử dụng vôn kế khung quay hoặc ampe kế khung quay để đo điện áp hay cường độ dòng điện xoay chiều. 
\end{itemize}}
\end{dang*}
