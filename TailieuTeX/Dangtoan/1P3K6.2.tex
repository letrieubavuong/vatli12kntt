\begin{dang}{LỰC MA SÁT}
\begin{itemize}
\item Biểu thức của lực ma sát trượt: $F_{mst}=\mu_t .N$
\item Biểu thức của lực ma sát lăn: $F_{msl}=\mu_\ell .N$
\item Phương pháp giải toán:
\begin{itemize}
\item Dùng phương pháp động lực học chiếu các lực tác dụng lên vật theo hai phương Ox và Oy (thường là phương chuyển động và phương vuông góc với chuyển động)
\item Giải hệ phương trình tìm áp lực N và các lực ma sát tương ứng
\item Kết hợp các công thức về động học chất điểm
\end{itemize}
\end{itemize}
\end{dang}
