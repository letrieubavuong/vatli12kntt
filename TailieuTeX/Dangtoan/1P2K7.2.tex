\begin{dang}{CHUYỂN ĐỘNG NÉM NGANG}
\begin{itemize}
\item Nếu từ cùng một độ cao, đồng thời ném ngang các vật khác nhau với các vận tốc khác nhau thì chúng đều rơi xuống đất cùng một lúc.
\item Nếu từ cùng một độ cao đồng thời ném các vật khác nhau với vận tốc khác nhau thì vật nào có vận tốc ném lớn hơn sẽ có tầm xa lớn hơn.
\item Nếu từ các độ cao khác nhau ném ngang các vật với cùng vận tốc thì vật nào được ném ở độ cao lớn hơn sẽ có tầm xa lớn hơn.
\item Tốc độ của vật theo từng phương: $v_x=v_0;\ v_y=gt$
\item Tốc độ của vật tại thời điểm t: $v=\sqrt{v_x^2+v_y^2}=\sqrt{v_0^2+\left(gt\right)^2}$ 
\end{itemize}
\end{dang}
