\begin{dang}{Cấu tạo hạt nhân}
\Anlythuyet{
\begin{itemize}
\item Xác định số nuclôn, số proton và số neutron có trong m gam hạt nhân 
\begin{itemize}
\item Số nuclôn: \boder{N_{\text{nuclôn}} = A.\dfrac{m}{M}.{N_A}}
\item Số proton: \boder{N_{\text{proton}} = Z.\dfrac{m}{A}.{N_A}}
\item Số neutron: \boder{N_{\text{neutron}} = \left({A - Z} \right).\dfrac{m}{A}.{N_A}}
\end{itemize}
\item Xác định các thông số liên quan đến hạt nhân
\begin{itemize}
\item Điện tích hạt nhân: \boder{Q = Ze}
\item Khối lượng riêng hạt nhân: \boder{D = \dfrac{m}{V}}
\item Mật độ điện tích hạt nhân: \boder{\rho = \dfrac{Q}{V}}
\end{itemize}
\end{itemize}
}
\end{dang}
