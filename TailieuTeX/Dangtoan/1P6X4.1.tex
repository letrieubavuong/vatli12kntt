\begin{dang*}{Lí thuyết}
\Anlythuyet{\begin{itemize}
\item Trong bộ phận phát động, tác nhân giãn nở sinh công.
\item Nguồn lạnh nhận nhiệt lượng do tác nhân toả ra để giảm nhiệt độ của động cơ.
\item Quá trình thuận nghịch là quá trình mà vật có thể tự quay về trạng thái ban đầu mà không cần có sự can thiệp của các vật khác.
\end{itemize}}	
\end{dang*}
