\begin{dang}{Bài tập phương trình trạng thái}
	\Anlythuyet{\begin{itemize}
			\item Đối với các bài toán đồ thị cần phân tích rõ các quá trình biến đổi, tóm tắt từng giai đoạn.
			\item Với các bài toán khí lý tưởng biến đổi qua nhiều giai đoạn, chúng ta chia nhỏ các giai đoạn để áp dụng biểu thức của các định luật chất khí để giải nhanh kết quả bài toán.
	\end{itemize}}
\end{dang}
