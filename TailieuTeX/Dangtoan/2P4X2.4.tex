\begin{dang}{Từ trường của dòng điện chạy trong ống dây dẫn hình trụ}
\Anlythuyet{\begin{itemize}
	\item Nếu sợi dây có đường kính d quấn sít nhau tạo thành ống thì số vòng dây của ống được tính:
	\begin{align*}
		\boder{N=\dfrac{\ell}{d}}
	\end{align*}
	\item Nếu dùng sợi dây có chiều dài $L$ để quấn ống thì số vòng dây của ống là: 
	\begin{align*}
		\boder{N=\dfrac{L}{2\pi R}}\ \text{với R là bán kính ống dây} 
	\end{align*}
\end{itemize}}
\end{dang}
